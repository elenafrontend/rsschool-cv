Code example

Remove all vowels from the string

          
            function disemvowel(str) {

              const vowels = ['a', 'e', 'i', 'o', 'u']
              let resultStr = str;
                
              for(let i = 0; i < str.length; i++) {
                vowels.forEach((vowel) => {
                  if (str[i].toLowerCase() === vowel) {
                    resultStr = resultStr.replace(str[i], '')
                  }
                })
              }
              return resultStr;
            }
